from typing import List
from pydantic_settings import BaseSettings
from pydantic import field_validator


class Settings(BaseSettings):
    ENV: str = "dev"
    API_TITLE: str = "Data API"
    API_DESCRIPTION: str = "Read-only data API with filtering and search capabilities."
    API_VERSION: str = "1.0.0"
    API_PREFIX: str = "/api"
    API_PASSWORD: str = "changeme"
    DB_PATH: str = "./app.db"
    HOST: str = "0.0.0.0"
    PORT: int = 5000
    CORS_ORIGINS: List[str] = ["*"]
    SHOW_DOCS: bool = True
    DEFAULT_PAGE_SIZE: int = 100
    MAX_PAGE_SIZE: int = 1000

    @field_validator("ENV")
    @classmethod
    def validate_env(cls, v: str) -> str:
        if v not in ("dev", "prod"):
            raise ValueError("ENV must be 'dev' or 'prod'")
        return v

    @property
    def is_dev(self) -> bool:
        return self.ENV == "dev"

    @property
    def is_prod(self) -> bool:
        return self.ENV == "prod"

    model_config = {"env_file": ".env", "env_file_encoding": "utf-8", "extra": "ignore"}


settings = Settings()

def _connect(path: str) -> sqlite3.Connection:
    conn = sqlite3.connect(path, check_same_thread=False)
    conn.row_factory = sqlite3.Row
    conn.execute("PRAGMA journal_mode=WAL")
    conn.execute("PRAGMA foreign_keys=ON")
    return conn


def get_connection(db_key: str = "default") -> sqlite3.Connection:
    path = DB_REGISTRY[db_key]
    return _connect(path)

SCHEMA_SQL = """
CREATE TABLE IF NOT EXISTS categories (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    name        TEXT    NOT NULL,
    slug        TEXT    NOT NULL UNIQUE,
    description TEXT,
    parent_id   INTEGER REFERENCES categories(id),
    created_at  TEXT    DEFAULT (datetime('now'))
);

CREATE TABLE IF NOT EXISTS customers (
    id           INTEGER PRIMARY KEY AUTOINCREMENT,
    name         TEXT NOT NULL,
    email        TEXT UNIQUE,
    phone        TEXT,
    city         TEXT,
    country      TEXT,
    country_code TEXT,
    created_at   TEXT DEFAULT (datetime('now'))
);

CREATE TABLE IF NOT EXISTS products (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    name        TEXT    NOT NULL,
    description TEXT,
    sku         TEXT    UNIQUE,
    category_id INTEGER REFERENCES categories(id),
    price       REAL    NOT NULL DEFAULT 0,
    stock_qty   INTEGER NOT NULL DEFAULT 0,
    active      INTEGER NOT NULL DEFAULT 1,
    created_at  TEXT    DEFAULT (datetime('now'))
);

CREATE TABLE IF NOT EXISTS orders (
    id           INTEGER PRIMARY KEY AUTOINCREMENT,
    customer_id  INTEGER REFERENCES customers(id),
    status       TEXT NOT NULL DEFAULT 'pending',
    total_amount REAL NOT NULL DEFAULT 0,
    currency     TEXT NOT NULL DEFAULT 'USD',
    notes        TEXT,
    created_at   TEXT DEFAULT (datetime('now')),
    updated_at   TEXT DEFAULT (datetime('now'))
);

CREATE TABLE IF NOT EXISTS order_items (
    id         INTEGER PRIMARY KEY AUTOINCREMENT,
    order_id   INTEGER NOT NULL REFERENCES orders(id),
    product_id INTEGER REFERENCES products(id),
    quantity   INTEGER NOT NULL DEFAULT 1,
    unit_price REAL    NOT NULL,
    subtotal   REAL    NOT NULL,
    created_at TEXT    DEFAULT (datetime('now'))
);
"""

def get_table_entry(table_name: str) -> Dict[str, Any]:
    if table_name not in TABLE_REGISTRY:
        raise KeyError(f"Unknown table: '{table_name}'")
    return TABLE_REGISTRY[table_name]

from fastapi import HTTPException, Security, status
from fastapi.security import APIKeyHeader, APIKeyQuery
from config import settings

_header_scheme = APIKeyHeader(name="X-API-Key", auto_error=False)
_query_scheme = APIKeyQuery(name="api_key", auto_error=False)


async def require_api_key(
    header_key: str = Security(_header_scheme),
    query_key: str = Security(_query_scheme),
) -> str:
    key = header_key or query_key
    if not key or key != settings.API_PASSWORD:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Invalid or missing API key.",
            headers={"WWW-Authenticate": "ApiKey"},
        )
    return key

from __future__ import annotations
import math
import sqlite3
from typing import Any, Dict, List, Optional
from fastapi import APIRouter, Depends, HTTPException, Query, Request
from auth import require_api_key
from config import settings
from database import get_connection
from registry import TABLE_REGISTRY, get_table_entry

router = APIRouter(prefix="", tags=["data"], dependencies=[Depends(require_api_key)])


def _row_to_dict(row: sqlite3.Row) -> Dict[str, Any]:
    return dict(row)


def _get_columns(conn: sqlite3.Connection, table: str) -> List[str]:
    cur = conn.execute(f"PRAGMA table_info({table})")
    return [r["name"] for r in cur.fetchall()]


def _count(conn: sqlite3.Connection, table: str, where: str, params: list) -> int:
    sql = f"SELECT COUNT(*) as n FROM {table}"
    if where:
        sql += f" WHERE {where}"
    row = conn.execute(sql, params).fetchone()
    return row["n"] if row else 0


def _build_where(entry, filters, search):
    conditions: List[str] = []
    params: List[Any] = []
    if search:
        searchable = entry.get("searchable_cols", [])
        if searchable:
            like_parts = [f"LOWER(CAST({c} AS TEXT)) LIKE ?" for c in searchable]
            conditions.append(f"({' OR '.join(like_parts)})")
            params.extend([f"%{search.lower()}%"] * len(searchable))
    for col, val in filters.items():
        if col in entry.get("filterable_cols", []):
            conditions.append(f"{col} = ?")
            params.append(val)
    return " AND ".join(conditions), params


def _query_page(conn, table, where, params, sort, page, page_size):
    total = _count(conn, table, where, params)
    pages = math.ceil(total / page_size) if page_size else 1
    offset = (page - 1) * page_size
    sql = f"SELECT * FROM {table}"
    if where:
        sql += f" WHERE {where}"
    sql += f" ORDER BY {sort} LIMIT ? OFFSET ?"
    rows = conn.execute(sql, params + [page_size, offset]).fetchall()
    return total, pages, [_row_to_dict(r) for r in rows]


@router.get("/tables", summary="List available tables")
def list_tables():
    result = []
    for name, entry in TABLE_REGISTRY.items():
        db_key = entry.get("db", "default")
        conn = get_connection(db_key)
        try:
            columns = _get_columns(conn, name)
            count = _count(conn, name, "", [])
        finally:
            conn.close()
        result.append({
            "name": name, "db": db_key, "row_count": count,
            "columns": columns, "searchable_columns": entry.get("searchable_cols", []),
        })
    return result


@router.get("/{table}", summary="Get all rows from a table")
def get_table(
    table: str,
    page: int = Query(1, ge=1),
    page_size: int = Query(settings.DEFAULT_PAGE_SIZE, ge=1, le=settings.MAX_PAGE_SIZE),
    request: Request = None,
):
    try:
        entry = get_table_entry(table)
    except KeyError:
        raise HTTPException(status_code=404, detail=f"Table '{table}' not found.")
    filters = {k: v for k, v in (request.query_params.items() if request else {})
               if k in entry.get("filterable_cols", [])}
    conn = get_connection(entry.get("db", "default"))
    try:
        where, params = _build_where(entry, filters, None)
        total, pages, items = _query_page(conn, table, where, params, entry.get("default_sort", "id"), page, page_size)
    finally:
        conn.close()
    return {"total": total, "page": page, "page_size": page_size, "pages": pages, "items": items}


@router.get("/{table}/search", summary="Search / filter a table")
def search_table(
    table: str,
    q: Optional[str] = Query(None),
    page: int = Query(1, ge=1),
    page_size: int = Query(settings.DEFAULT_PAGE_SIZE, ge=1, le=settings.MAX_PAGE_SIZE),
    request: Request = None,
):
    try:
        entry = get_table_entry(table)
    except KeyError:
        raise HTTPException(status_code=404, detail=f"Table '{table}' not found.")
    filters = {k: v for k, v in (request.query_params.items() if request else {})
               if k in entry.get("filterable_cols", []) and k != "q"}
    conn = get_connection(entry.get("db", "default"))
    try:
        where, params = _build_where(entry, filters, q)
        total, pages, items = _query_page(conn, table, where, params, entry.get("default_sort", "id"), page, page_size)
    finally:
        conn.close()
    return {"total": total, "page": page, "page_size": page_size, "pages": pages, "items": items}


@router.get("/{table}/{id}", summary="Get a single row by ID")
def get_row(table: str, id: int):
    try:
        entry = get_table_entry(table)
    except KeyError:
        raise HTTPException(status_code=404, detail=f"Table '{table}' not found.")
    conn = get_connection(entry.get("db", "default"))
    try:
        row = conn.execute(f"SELECT * FROM {table} WHERE id = ?", [id]).fetchone()
    finally:
        conn.close()
    if row is None:
        raise HTTPException(status_code=404, detail=f"Row {id} not found in '{table}'.")
    return _row_to_dict(row)

def seed(clear: bool = False):
    conn = get_connection("default")
    if clear:
        for t in ["order_items", "orders", "products", "customers", "categories"]:
            conn.execute(f"DROP TABLE IF EXISTS {t}")
        conn.commit()
        print("Dropped all tables.")
    conn.executescript(SCHEMA_SQL)
    conn.commit()
    if not clear and conn.execute("SELECT COUNT(*) FROM categories").fetchone()[0] > 0:
        print("Already seeded. Use --clear to reseed.")
        conn.close()
        return
    cat_ids = []
    for name, slug, desc in CATEGORIES:
        cur = conn.execute("INSERT INTO categories (name,slug,description) VALUES (?,?,?)", (name,slug,desc))
        cat_ids.append(cur.lastrowid)
    cust_ids = []
    for name, email, phone, city, country, cc in CUSTOMERS:
        cur = conn.execute("INSERT INTO customers (name,email,phone,city,country,country_code) VALUES (?,?,?,?,?,?)", (name,email,phone,city,country,cc))
        cust_ids.append(cur.lastrowid)
    prod_data = []
    for name, sku, cat_idx, price, stock in PRODUCTS:
        cur = conn.execute("INSERT INTO products (name,sku,category_id,price,stock_qty,active) VALUES (?,?,?,?,?,1)", (name,sku,cat_ids[cat_idx],price,stock))
        prod_data.append({"id": cur.lastrowid, "price": price})
    base = datetime.utcnow() - timedelta(days=90)
    for _ in range(50):
        cust_id = random.choice(cust_ids)
        status  = random.choice(STATUSES)
        created = (base + timedelta(days=random.randint(0, 90))).strftime("%Y-%m-%d %H:%M:%S")
        cur = conn.execute("INSERT INTO orders (customer_id,status,currency,created_at,updated_at) VALUES (?,?,?,?,?)", (cust_id,status,"USD",created,created))
        order_id = cur.lastrowid
        total = 0.0
        for _ in range(random.randint(1, 4)):
            prod = random.choice(prod_data)
            qty  = random.randint(1, 5)
            sub  = round(prod["price"] * qty, 2)
            total += sub
            conn.execute("INSERT INTO order_items (order_id,product_id,quantity,unit_price,subtotal,created_at) VALUES (?,?,?,?,?,?)", (order_id,prod["id"],qty,prod["price"],sub,created))
        conn.execute("UPDATE orders SET total_amount=? WHERE id=?", (round(total,2), order_id))
    conn.commit()
    conn.close()
    print("Seeded: categories, customers, products, orders, order_items.")

import logging
from contextlib import asynccontextmanager
import uvicorn
from fastapi import FastAPI, Request, status
from fastapi.middleware.cors import CORSMiddleware
from fastapi.responses import JSONResponse
from config import settings
from database import DB_REGISTRY, get_connection
from models import SCHEMA_SQL
from routers import router

logging.basicConfig(
    level=logging.DEBUG if settings.is_dev else logging.INFO,
    format="%(asctime)s  %(levelname)-8s  %(name)s – %(message)s",
)
log = logging.getLogger(__name__)


@asynccontextmanager
async def lifespan(app: FastAPI):
    for db_key, db_path in DB_REGISTRY.items():
        conn = get_connection(db_key)
        try:
            conn.executescript(SCHEMA_SQL)
            conn.commit()
            log.info("DB '%s' (%s) ready.", db_key, db_path)
        finally:
            conn.close()
    yield
    log.info("Shutting down.")


def create_app() -> FastAPI:
    app = FastAPI(
        title=settings.API_TITLE,
        description=settings.API_DESCRIPTION,
        version=settings.API_VERSION,
        docs_url="/docs" if settings.SHOW_DOCS else None,
        redoc_url="/redoc" if settings.SHOW_DOCS else None,
        openapi_url="/openapi.json",
        lifespan=lifespan,
    )
    app.add_middleware(
        CORSMiddleware,
        allow_origins=settings.CORS_ORIGINS,
        allow_credentials=True,
        allow_methods=["GET", "OPTIONS"],
        allow_headers=["*"],
    )

    @app.exception_handler(Exception)
    async def _unhandled(request: Request, exc: Exception):
        log.exception("Unhandled error: %s", exc)
        return JSONResponse(status_code=status.HTTP_500_INTERNAL_SERVER_ERROR, content={"detail": "Internal server error."})

    @app.get("/health", tags=["system"], summary="Health check")
    def health():
        return {"status": "ok", "env": settings.ENV, "version": settings.API_VERSION}

    app.include_router(router, prefix=settings.API_PREFIX)
    return app


app = create_app()

if __name__ == "__main__":
    uvicorn.run("main:app", host=settings.HOST, port=settings.PORT, reload=settings.is_dev, log_level="debug" if settings.is_dev else "info")
